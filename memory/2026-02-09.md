### Exercise Programming Defaults
```dart
// Added _getRestSecondsWithDefaults() in exercise_programming.dart
static int _getRestSecondsWithDefaults(int restSeconds, String exerciseRef) {
  if (restSeconds != 60) return restSeconds;
  
  final exerciseType = _detectExerciseType(exerciseRef);
  switch (exerciseType) {
    case 'kettlebell': return 90;
    case 'goblet_squat': return 75;
    case 'dumbbell': return 60;
    // ... other exercise types
    default: return 60;
  }
}
```
